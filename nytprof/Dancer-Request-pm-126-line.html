500sub _parse_get_params {
501 my ($self) = @_;
502 return $self->{_query_params} if defined $self->{_query_params};
503 $self->{_query_params} = {};
504
505 my $source = $self->env->{QUERY_STRING} || '';
506 foreach my $token (split /[&;]/, $source) {
507 my ($key, $val) = split(/=/, $token, 2);
508 next unless defined $key;
509 $val = (defined $val) ? $val : '';
510 $key = $self->_url_decode($key);
511 $val = $self->_url_decode($val);
512
513 # looking for multi-value params
514 if (exists $self->{_query_params}{$key}) {
515 my $prev_val = $self->{_query_params}{$key};
516 if (ref($prev_val) && ref($prev_val) eq 'ARRAY') {
517 push @{$self->{_query_params}{$key}}, $val;
518 }
519 else {
520 $self->{_query_params}{$key} = [$prev_val, $val];
521 }
522 }
523
524 # simple value param (first time we see it)
525 else {
526 $self->{_query_params}{$key} = $val;
527 }
528 }
529 return $self->{_query_params};
530}

534

555sub _read {
556 my ($self,) = @_;
557 my $remaining = $self->content_length - $self->{_read_position};
558 my $maxlength = $self->{_chunk_size};
559
560 return if ($remaining <= 0);
561
562 my $readlen = ($remaining > $maxlength) ? $maxlength : $remaining;
563 my $buffer;
564 my $rc;
565
566 $rc = $self->input_handle->read($buffer, $readlen);
567
568 if (defined $rc) {
569 $self->{_read_position} += $rc;
570 return $buffer;
571 }
572 else {
573 raise core_request => "Unknown error reading input: $!";
574 }
